```python
# src/da_facts.py
from __future__ import annotations
from typing import Any

PPE_DETAIL_FIELDS = ("gross", "accum_dep", "impairment", "net",
                     "period_increase", "period_decrease", "period_dep")

def validate_da_facts(facts: dict[str, Any]) -> list[str]:
    errors: list[str] = []
    if "base_year" not in facts:
        errors.append("base_year missing")
    ppe_detail = facts.get("ppe_detail", {})
    missing_flags = {(m["year"], m["category"], m["field"])
                     for m in facts.get("missing_flags", [])}
    for year, cats in ppe_detail.items():
        for cat, vals in cats.items():
            if not isinstance(vals, dict):
                continue
            for field in PPE_DETAIL_FIELDS:
                if field not in vals:
                    continue
                v = vals[field]
                if v == 0.0 and (year, cat, field) not in missing_flags:
                    errors.append(f"{year}.{cat}.{field}=0 without missing_flag (zero-fill forbidden)")
    return errors
```

```python
from src.annual_report_utils import find_line, compact_window, find_all_lines

def locate_note_sections(lines: list[str]) -> dict[str, dict]:
    """定位固定资产/无形资产/在建工程附注 + 会计政策年限残值率表。"""
    sections: dict[str, dict] = {}
    # 会计政策:固定资产折旧年限表(新乳业 17、固定资产 ... 折旧年限 残值率 年折旧率)
    pol = find_line(lines, ["固定资产", "折旧年限"]) or find_line(lines, ["17、固定资产"])
    if pol is not None:
        sections["ppe_policy"] = compact_window(lines, pol, before=2, after=60)
    intang_pol = find_line(lines, ["无形资产", "摊销年限"]) or find_line(lines, ["21、无形资产"])
    if intang_pol is not None:
        sections["intangible_policy"] = compact_window(lines, intang_pol, before=2, after=40)
    # 附注明细:固定资产/在建工程/无形资产(财务报表附注节,非资产负债表合并行)
    # 用 find_all_lines 取所有出现,挑附注节(通常在"七、合并财务报表项目注释"后)
    for key, patterns in [("ppe_detail", ["固定资产"]),
                          ("cip_detail", ["在建工程"]),
                          ("intangible_detail", ["无形资产"])]:
        hits = find_all_lines(lines, patterns)
        # 选最长的窗口(附注明细表通常最长);细节留实现
        if hits:
            sections[key] = compact_window(lines, hits[len(hits)//2], before=5, after=120)
    return sections
```

```python
# src/da_facts.py  追加
from src.annual_report_utils import call_llm

def extract_note(note_type: str, window_text: str, year: int,
                 allowed_fields: set[str]) -> dict[str, Any]:
    schema_hint = {"categories": [{"name": str, **{f: float for f in allowed_fields}}]}
    messages = [{
        "role": "user",
        "content": f"从年报附注提取{note_type}(年份{year})。只填表不推算,抽不到留 null。"
                   f"严格按 schema:{schema_hint}。不要输出 schema 外字段。附注片段:\n{window_text}"
    }]
    raw = call_llm(messages)
    if raw.get("error"):
        return {"error": raw["error"], "categories": []}
    # 防脏守卫:剔除 schema 外字段
    for cat in raw.get("categories", []):
        for k in list(cat.keys()):
            if k != "name" and k not in allowed_fields:
                del cat[k]
    return raw
```

```python
# src/da_facts.py  追加
TOLERANCE_MILLION = 1.0

def check_rollforward(year, category, vals: dict) -> dict:
    opening = vals.get("opening_net", 0.0) or 0.0
    increase = vals.get("period_increase", 0.0) or 0.0
    dep = vals.get("period_dep", 0.0) or 0.0
    decrease = vals.get("period_decrease", 0.0) or 0.0
    impair = vals.get("impairment", 0.0) or 0.0
    closing = vals.get("closing_net", 0.0) or 0.0
    calc = opening + increase - dep - decrease - impair
    residual = calc - closing
    return {"year": year, "category": category, "closed": abs(residual) < TOLERANCE_MILLION,
            "residual": residual}
```

```python
from src.company_paths import recon_dir

def extract_company_facts(company_dir, base_year: int, years: list[int]) -> dict:
    jobs = [(nt, y) for nt in ("ppe_detail","cip_detail","intangible_detail") for y in years]
    # + 会计政策(仅最新年)
    jobs.append(("ppe_policy", base_year))
    jobs.append(("intangible_policy", base_year))

    def run(job):
        nt, year = job
        md = annual_markdown_path(company_dir, str(year))
        if not md:
            return {"note_type": nt, "year": year, "error": "md not found"}
        lines = md.read_text(encoding="utf-8").splitlines()
        sections = locate_note_sections(lines)
        sec = sections.get(nt)
        if not sec:
            return {"note_type": nt, "year": year, "error": "section not located"}
        allowed = {"gross","accum_dep","impairment","net","period_increase",
                   "period_decrease","period_dep","opening_net","closing_net"}
        return {"note_type": nt, "year": year,
                "data": extract_note(nt, sec["text"], year, allowed)}

    results = parallel_map(run, jobs, max_workers=3)  # fallback 降并发避 429
    facts = {"company": company_dir.name, "base_year": base_year,
             "ppe_detail": {}, "cip_detail": {}, "intangible_detail": {},
             "policy": {}, "roll_forward_checks": [], "missing_flags": [],
             "evidence_anchors": []}
    for r in results:
        # 合并 r["data"] 到 facts 对应节(细节留实现)
        pass
    facts["validation_errors"] = validate_da_facts(facts)
    out = recon_dir(company_dir)
    write_json(out / "da_facts_latest.json", facts)
    return facts
```

```python
# src/da_roll.py
from __future__ import annotations
from pathlib import Path
from typing import Any
import yaml

class DaAlignError(RuntimeError):
    pass

def load_da_schedule(path: Path, defaults_base_period: str) -> dict | None:
    if not path.exists():
        return None
    sched = yaml.safe_load(path.read_text(encoding="utf-8")) or {}
    if not sched.get("enabled", False):
        return None
    base_year = sched.get("base_year")
    if str(base_year) != str(defaults_base_period)[:4]:
        raise DaAlignError(
            f"da_schedule.base_year={base_year} != defaults.base_period={defaults_base_period}")
    return sched
```

```python
# src/da_roll.py  追加
from dataclasses import dataclass

@dataclass
class Cohort:
    gross: float
    salvage_rate: float
    life: int
    start_year: int
    def annual_dep(self) -> float:
        return self.gross * (1.0 - self.salvage_rate) / self.life
    def dep_in_year(self, year: int) -> float:
        if year < self.start_year:
            return 0.0
        elapsed = year - self.start_year
        if elapsed >= self.life:
            return 0.0
        return self.annual_dep()
    def net_in_year(self, year: int) -> float:
        if year < self.start_year:
            return 0.0
        elapsed = min(year - self.start_year, self.life)
        return max(self.gross - self.annual_dep() * elapsed - self.gross * self.salvage_rate * (1 if elapsed >= self.life else 0), self.gross * self.salvage_rate)
```

```python
    pass

@dataclass
class CipState:
    base_cip: float
    base_transfers: dict[int, float]      # year -> amount
    expansion_capex: dict[int, float]
    expansion_transfers: dict[int, float]
    life: int
    salvage: float
    start_year: int
    def cip_balance(self, year: int) -> float:
        bal = self.base_cip
        for y in range(self.start_year, year + 1):
            bal += self.expansion_capex.get(y, 0.0)
            bal -= self.base_transfers.get(y, 0.0)
            bal -= self.expansion_transfers.get(y, 0.0)
        if bal < -1e-6:
            raise CipInvariantError(f"cip negative at {year}: {bal}")
        return bal
    def transferred_cohorts(self, year: int) -> list[Cohort]:
        out = []
        amt = self.base_transfers.get(year, 0.0) + self.expansion_transfers.get(year, 0.0)
        if amt > 0:
            out.append(Cohort(amt, self.salvage, self.life, year))
        return out

def roll_cip(base_cip, base_cip_to_fixed, expansion_capex_by_year,
             expansion_cip_to_fixed, cat_life, cat_salvage, start_year):
    # 不变量:累计 base 转固 ≤ base cip
    cum_base = sum(base_cip_to_fixed.values())
    if cum_base > base_cip + 1e-6:
        raise CipInvariantError(f"base cip over-transferred: {cum_base} > {base_cip}")
    return CipState(base_cip, base_cip_to_fixed, expansion_capex_by_year,
                    expansion_cip_to_fixed, cat_life, cat_salvage, start_year)
```

```python
def _calibrate_scale(cats, base_reported_dep):
    """存量稳态折旧缩放匹配 base 年披露 depr_fa_coga_dpba,否则 light/heavy EBIT 不连续。"""
    policy_dep = sum(c["base_gross"] * (1 - c["salvage_rate"]) / c["life_years"] for c in cats)

        return 1.0
    return base_reported_dep / policy_dep  # flag 残差过大留实现(>20% warning)
```
